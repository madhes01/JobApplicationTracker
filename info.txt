# 3. Install pg explicitly if needed

# 4. Create .env.local and paste your Neon connection string

# 5. Run DB migration
pnpm dlx @better-auth/cli@latest migrate --config lib/auth/auth.ts

# 6. Start dev server
pnpm dev
